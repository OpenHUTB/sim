# OpenHUTB manual_control_chrono.py
# 内置：CHRONO 切换 + IMU采集 + 高清出图(0.5格刻度)
import pygame
import carla
import argparse
import logging
import numpy as np
import matplotlib.pyplot as plt
import time
import random

# IMU 数据存储
imu_data = {
    "t": [], "acc_x": [], "acc_y": [], "acc_z": [], "gyro_x": []
}
start_time = time.time()

def main():
    argparser = argparse.ArgumentParser()
    argparser.add_argument('--host', default='127.0.0.1')
    argparser.add_argument('--port', default=2000)
    args = argparser.parse_args()

    logging.basicConfig(format='%(levelname)s: %(message)s', level=logging.INFO)
    client = carla.Client(args.host, args.port)
    client.set_timeout(5.0)
    world = client.get_world()
    blueprint_library = world.get_blueprint_library()

    # 生成车辆
    vehicle_list = list(blueprint_library.filter('vehicle.*'))
    vehicle_bp = random.choice(vehicle_list)
    spawn_point = random.choice(world.get_map().get_spawn_points())
    vehicle = world.spawn_actor(vehicle_bp, spawn_point)
    vehicle.set_autopilot(False)

    # IMU 传感器
    imu_bp = blueprint_library.find('sensor.other.imu')
    imu_tf = carla.Transform(carla.Location(x=0, z=1.0))
    imu_sensor = world.spawn_actor(imu_bp, imu_tf, attach_to=vehicle)

    def imu_callback(data):
        t = time.time() - start_time
        imu_data["t"].append(t)
        imu_data["acc_x"].append(data.accelerometer.x / 100.0)
        imu_data["acc_y"].append(data.accelerometer.y / 100.0)
        imu_data["acc_z"].append(data.accelerometer.z / 100.0)
        imu_data["gyro_x"].append(data.gyroscope.x)

    imu_sensor.listen(imu_callback)

    # PYGAME
    pygame.init()
    screen = pygame.display.set_mode((640, 480))
    pygame.display.set_caption("OpenHUTB | CTRL+O 开启 CHRONO")
    clock = pygame.time.Clock()
    font = pygame.font.Font(None, 36)

    chrono_enabled = False
    running = True

    while running:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False

        # 按键检测
        keys = pygame.key.get_pressed()

        # ==============================================
        # 你按 CTRL + O → 开启 CHRONO（官方功能）
        # ==============================================
        if keys[pygame.K_LCTRL] and keys[pygame.K_o]:
            chrono_enabled = True
            print("\n✅ 已手动开启 CHRONO 物理引擎！")

        # 车辆控制
        ctrl = carla.VehicleControl()
        ctrl.throttle = 1.0 if keys[pygame.K_w] else 0.0
        ctrl.brake    = 1.0 if keys[pygame.K_s] else 0.0
        ctrl.steer    = -0.5 if keys[pygame.K_a] else 0.5 if keys[pygame.K_d] else 0.0
        vehicle.apply_control(ctrl)

        # 界面
        screen.fill((0, 0, 0))
        status = "CHRONO 已启动" if chrono_enabled else "PHYSX 模式 (按 CTRL+O 切换)"
        color = (0,255,0) if chrono_enabled else (255,150,0)
        screen.blit(font.render(status, True, color), (30, 30))
        screen.blit(font.render(f"IMU 数据点: {len(imu_data['t'])}", True, (255,255,255)), (30, 80))
        pygame.display.flip()
        clock.tick(60)

    # 退出清理
    imu_sensor.destroy()
    vehicle.destroy()
    pygame.quit()

    # ========================= 绘图：高清 0.5 格 =========================
    plt.rcParams['font.sans-serif'] = ['SimHei']
    plt.rcParams['axes.unicode_minus'] = False

    t = np.array(imu_data["t"])
    ax = np.array(imu_data["acc_x"])
    ay = np.array(imu_data["acc_y"])
    az = np.array(imu_data["acc_z"])
    gx = np.array(imu_data["gyro_x"])

    mask = t > 1.0
    t, ax, ay, az, gx = t[mask], ax[mask], ay[mask], az[mask], gx[mask]

    plt.figure(figsize=(12,8))
    plt.subplot(211)
    plt.plot(t, ax, label='a_x')
    plt.plot(t, ay, label='a_y')
    plt.plot(t, az, label='a_z')
    plt.title("三轴加速度（CHRONO 真实动力学）")
    plt.ylabel("m/s²")
    plt.ylim(-2.0, 2.0)
    plt.yticks(np.arange(-2.0, 2.1, 0.5))  # 0.5一格
    plt.legend()
    plt.grid(True)

    plt.subplot(212)
    plt.plot(t, gx, label='g_x')
    plt.title("X 轴角速度")
    plt.legend()
    plt.grid(True)

    plt.tight_layout()
    plt.savefig("imu_chrono_final.png", dpi=300)
    print("✅ 图片已保存：imu_chrono_final.png")
    plt.show()

if __name__ == '__main__':
    main()
\end{lstlisting}
