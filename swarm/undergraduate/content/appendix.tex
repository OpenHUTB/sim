\begin{lstlisting}[style=hutbcode,language=bash,caption={AirSim优化模型编译与启用流程},label={lst:airsim-opt-build}]
build_release.bat
copy /Y AirSim.dll <AirSimPluginDir>\Binaries\Win64\AirSim.dll
UnrealEditor.exe <Project>.uproject
AirSim.exe -allow-thread-optimize
\end{lstlisting}
